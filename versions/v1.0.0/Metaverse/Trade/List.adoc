===== List

List indicates an action of selling in-game items for trading.

===== Interface

[,typescript]
----

interface List {
    action: string;
    address?: string;
    id?: string;
    name?: string;
    standard?: string;
    symbol?: string;
    value: string;
    uri?: string;
    parsed_image_url?: string;
    cost: {
        name?: string;
        symbol?: string;
        decimals?: number;
        value: string;
        address?: string;
        standard?: string;
    };
}
----
