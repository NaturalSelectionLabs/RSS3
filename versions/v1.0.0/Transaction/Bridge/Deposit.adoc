===== Deposit

===== Interface

[,typescript]

----
interface Deposit {
    action: string;
    source_network: string;
    target_network: string;
    token: {
        address?: string;
        id?: string;
        value?: string;
        name?: string;
        symbol?: string;
        uri?: string;
        decimals?: number;
        standard?: string;
    }
}
